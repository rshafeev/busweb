 13 /**

 17 cityways.page.MainPage = cityways.Class({
 18 
 19 	/**
 20 	 * [mapWidget description]
 21 	 * @type {cityways.Map}
 22 	 */
 23 	mapWidget : null,
 24 
 25 	/**
 26 	 * Видимость правой панели.
 27 	 * @private
 28 	 * @fieldOf cityways.page.MainPage.prototype
 29 	 * @default false
 30 	 * @typedef {bool}
 31 	 * @type {bool}

 33 	rightPanelVisible : null,
 34 	
 35 	/**
 36 	 * cityways.Pages.MainPage.RouteTypesPanel Панель "Виды транспорта"
 37 	 */
 38 	settingsPanel : null,
 39 
 40 	widgetEventHandlers : null,
 41 
 42 	/**
 43 	 * @private
 44 	 * @type {Object}
 45 	 * @example Структура:
 46 	 * {
 47 	 * scale : {Number},
 48 	 * location : {
 49 	 * 			     lat : {Number},
 50 	 * 			     lon : {Number}	
 51 	 * 			  }
 52 	 * }
 53 	 */
 54 	currentCity : null,
 55 
 56 	/**
 57 	 * [initialize description]
 58 	 * @constructor
 59 	 * @this cityways.page.MainPage
 60 	 * @class cityways.page.MainPage
 61 	 * @param  {[type]} options [description]
 62 	 * @return {[type]}         [description]
 63 	 */
 64 	initialize : function(options) {
 65 		var self = this;
 66 		selectbox_initialize();
 67 
 68 		cityways.helper.styles.includeCSSFile(cityways.options.getResourcePath() + "css/",
 69 												[{
 70 													name : "main.css"
 71 												}]);
 72 		this._prepareTabs();
 73 		this.currentCity = options.currentCity;
 74 		this.widgetEventHandlers = new cityways.page.main.WidgetEventHandlers();
 75 		this.settingsPanel = new cityways.page.main.SettingsPanel({
 76 					availableRouteTypes : options.routeTypes
 77 				});
 78 
 79 		var mapOptions = {
 80 			mapProvider : "google",
 81 			zoom : this.currentCity.scale,
 82 			center : {
 83 				lat : this.currentCity.location.lat,
 84 				lng : this.currentCity.location.lon
 85 			}
 86 		};
 87 		this.mapWidget = new cityways.MapWidget("map_canvas",mapOptions);
 88 		
 89 		this.visibleRightPanel(false);
 90 		
 91 		$(window).bind("resize", function(e) {
 92             self.on_resize_window(e);
 93         });
 94 		$(document).ready(function() {
 95 		 	self.mapWidget.setHeight(cityways.util.getWindowSize().height - 134);
 96 		 });
 97 		
 98 		cityways.logger.info('MainPage was initialized!!');
 99 		
100 	},
101

127 	getPathsOptions : function() {
128 		var city = this.getCurrentCity();
129 		var isTransitions = true;
130 		var alg_type = "c_time";
131 		var usage_routeTypes = [];
132 
133 		if (isTransitions == true) {
134 			usage_routeTypes.push({
135 						discount : 1.0,
136 						route_type_id : "c_route_transition"
137 					});
138 		}
139 
140 		usage_routeTypes.push({
141 					discount : 1.0,
142 					route_type_id : "c_route_station_input"
143 				});
144 		usage_routeTypes.push({
145 					discount : 1.0,
146 					route_type_id : "c_route_station_output"
147 				});
148 		usage_routeTypes.push({
149 					discount : 0.5,
150 					route_type_id : "c_route_metro"
151 				});
152 		usage_routeTypes.push({
153 					discount : 1.0,
154 					route_type_id : "c_route_bus"
155 				});
156 		usage_routeTypes.push({
157 					discount : 1.0,
158 					route_type_id : "c_route_trolley"
159 				});
160 
161 		var opts = {
162 			cityID : city.id,
163 			p1 : {
164 				lat : 50.0253640226659,
165 				lon : 36.3350757963562
166 			},
167 			p2 : {
168 				lat : 50.0353179327227,
169 				lon : 36.2199825018311
170 			},
171 			outTime : {
172 				dayID : 'c_Sunday',
173 				timeStart : 1000 * (10 * 60 * 60 + 10 * 60)
174 			},
175 			algStrategy : alg_type,
176 			usageRouteTypes : usage_routeTypes
177 		};
178 		return opts;
179 
180 	},
181 
182 	/**
183 	 * 
184 	 * @param {bool}
185 	 *            true: show, false : hide
186 	 */
187 	visibleRightPanel : function(value) {
188 		 var ResourceURI = cityways.options.ResourceURI;
189 		if(this.rightPanelVisible == value)
190 			return;
191 		if (value == true) {
192 			$("#map_canvas").width('68%').css({
193 						cursor : "auto",
194 						backgroundColor : "rgb(226, 226, 226)"
195 					});
196 			
197 			//document.img.src = ResourceURI + 'themes/default/images/arrow_right.png';
198 					
199 		} else {
200 			$("#map_canvas").width('98.5%').css({
201 						cursor : "auto",
202 						backgroundColor : "rgb(226, 226, 226)"
203 					});
204 					
205 			//document.img.src =ResourceURI + 'themes/default/images/arrow_left.png';
206 		}
207 		this.rightPanelVisible = value;
208 		
209 	},
210 
211 	_prepareTabs : function() {
212 		$("div.selectTabs_first, div.selectTabs_second").each(function() {
213 		var tmp = $(this);
214 		// console.log($(tmp).find(" .lineTabs li"));
215 		$(tmp).find(".lineTabs li").each(function(i) {
216 					$(tmp).find(".lineTabs li:eq(" + i + ") a").click(
217 							function() {
218 								var tab_id = i + 1;
219 								$(tmp).find(".lineTabs li a")
220 										.removeClass("active");
221 								$(this).addClass("active");
222 								$(tmp).find(".content div").stop(false, false)
223 										.hide();
224 								$(tmp).find(".tab" + tab_id).stop(false, false)
225 										.show();
226 								return false;
227 							});
228 				});
229 		});
230 	},

237 				offsetX : -25
238 			});
239 	}
