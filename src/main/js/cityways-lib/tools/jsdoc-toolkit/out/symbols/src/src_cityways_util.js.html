 18  cityways.util = {
 19 
 20     _browserDetec : null,
 21     
 22     browserVersion : function(){
 23         if(cityways.util._browserDetec == null){
 24             cityways.util._browserDetec = new cityways.BrowserDetect();
 25         }
 26         return cityways.util._browserDetec.version;
 27     },
 28     
 29     browser : function(){
 30         if(cityways.util._browserDetec == null){
 31             cityways.util._browserDetec = new cityways.BrowserDetect();
 32         }
 33         return cityways.util._browserDetec.browser;
 34     },
 35 
 36 
 37    /**
 38    * Возвращает uri-путь к скрипту
 39    * @param  {String} scriptName Имя файла скрипта
 40    * @return {String}            URI путь к скрипту
 41    */
 42    getScriptLocation: function(scriptName) {
 43     var regPattern = "(^|(.*?\\/))("+scriptName+"[^\\/]*?\\.js)(\\?|$)";
 44     var r = new RegExp(regPattern),
 45     s = document.getElementsByTagName('script'),
 46     src, m, l = "";
 47     for(var i=0, len=s.length; i<len; i++) {
 48         src = s[i].getAttribute('src');
 49         if(src) {
 50             m = src.match(r);
 51             if(m) {
 52                 l = m[1];
 53                 break;
 54             }
 55         }
 56     }
 57     return l;
 58 },
 59 
 60     /**
 61      * [getWindowSize description]
 62      * @return {[type]} [description]
 63      */
 64      getWindowSize : function() {
 65         var myWidth = 0, myHeight = 0;
 66         if (typeof (window.innerWidth) == 'number') {
 67         // Non-IE
 68         myWidth = window.innerWidth;
 69         myHeight = window.innerHeight;
 70     } else if (document.documentElement
 71         && (document.documentElement.clientWidth || document.documentElement.clientHeight)) {
 72         // IE 6+ in 'standards compliant mode'
 73         myWidth = document.documentElement.clientWidth;
 74         myHeight = document.documentElement.clientHeight;
 75     } else if (document.body
 76         && (document.body.clientWidth || document.body.clientHeight)) {
 77         // IE 4 compatible
 78         myWidth = document.body.clientWidth;
 79         myHeight = document.body.clientHeight;
 80     }
 81     return {
 82         height : myHeight,
 83         width : myWidth
 84     };
 85 }
