 12  /**
 13  * @private
 14  * @namespace
 15  * @description Содержит функции логгирования
 16  */
 17  cityways.logger = {
 18 
 19  	DEBUG_LEVEL : false,
 20 
 21  	INFO_LEVEL  : false,  
 22 
 23  	WARN_LEVEL  : false,
 24  	
 25  	/**
 26  	 * [info description]
 27  	 */
 28  	info  : function(){},
 29 
 30  	/**
 31  	 * [debug description]
 32  	 */
 33  	debug : function(){},
 34 
 35  	/**
 36  	 * [warn description]
 37  	 */
 38  	warn  : function(){},
 39 
 40  	setLevelMode : function(level, val){
 41  		if(level == "INFO_LEVEL"){
 42  			if(val == true)
 43  				cityways.logger.info  = console.log.bind(console);
 44  			else
 45  				cityways.logger.info = function(){};
 46 		}
 47  		if(level == "DEBUG_LEVEL"){
 48  			if(val == true)
 49  				cityways.logger.debug  = console.info.bind(console);
 50  			else
 51  				cityways.logger.debug = function(){};
 52 		}
 53 		 if(level == "WARN_LEVEL"){
 54  			if(val == true)
 55  				cityways.logger.warn  = console.warn.bind(console);
 56  			else
 57  				cityways.logger.warn = function(){};
 58 		}
 59 		cityways.logger[level] = val;
 60  		
 61  	}
 62 
 63  };
