 10 /**
 11  * Статический класс
 12  */
 13 cityways.language = {
 14     
 15     /** 
 16      * Property: code
 17      * {String}  Current language code to use in OpenLayers.  Use the
 18      *     <setCode> method to set this value and the <getCode> method to
 19      *     retrieve it.
 20      */
 21     code: null,
 22 
 23     /** 
 24      * APIProperty: defaultCode
 25      * {String} Default language to use when a specific language can't be
 26      *     found.  Default is "en".
 27      */
 28     defaultCode: "ru",
 29         
 30     /**
 31      * APIFunction: getCode
 32      * Get the current language code.
 33      *
 34      * Returns:
 35      * {String} The current language code.
 36      */
 37     getCode: function() {
 38         if(!cityways.language.code) {
 39             cityways.language.setCode();
 40         }
 41         
 42 
 43 
 44         return cityways.language.code;
 45     },
 46     
 47     /**
 48      * APIFunction: setCode
 49      * Set the language code for string translation.  This code is used by
 50      *     the <OpenLayers.Lang.translate> method.
 51      *
 52      * Parameters:
 53      * code - {String} These codes follow the IETF recommendations at
 54      *     http://www.ietf.org/rfc/rfc3066.txt.  If no value is set, the
 55      *     browser's language setting will be tested.  If no <OpenLayers.Lang>
 56      *     dictionary exists for the code, the <OpenLayers.String.defaultLang>
 57      *     will be used.
 58      */
 59     setCode: function(code) {
 60         var lang;
 61         
 62         if(!code) {
 63             code = (cityways.BROWSER_NAME == "msie") ?
 64                 navigator.userLanguage : navigator.language;
 65         }
 66         var parts = code.split('-');
 67         parts[0] = parts[0].toLowerCase();
 68         
 69         if(typeof cityways.lang[parts[0]] == "object") {
 70             lang = parts[0];
 71         }
 72         
 73         // check for regional extensions
 74         if(parts[1]) {
 75             var testLang = parts[0] + '-' + parts[1].toUpperCase();
 76             if(typeof cityways.lang[testLang] == "object") {
 77                 lang = testLang;
 78             }
 79         }
 80         if(!lang) {
 81             cityways.Console.warn(
 82                 'Failed to find cityways.Lang.' + parts.join("-") +
 83                 ' dictionary, falling back to default language'
 84             );
 85             lang = cityways.Language.defaultCode;
 86         }
 87         
 88         cityways.language.code = lang;
 89     },
 90 
 91     /**
 92      * APIMethod: translate
 93      * Looks up a key from a dictionary based on the current language string.
 94      *     The value of <getCode> will be used to determine the appropriate
 95      *     dictionary.  Dictionaries are stored in <OpenLayers.Lang>.
 96      *
 97      * Parameters:
 98      * key - {String} The key for an i18n string value in the dictionary.
 99      * context - {Object} Optional context to be used with
100      *     <OpenLayers.String.format>.
101      * 
102      * Returns:
103      * {String} A internationalized string.
104      */
105     translate: function(key) {
106         var dictionary = cityways.lang[cityways.language.getCode()];
107         var message = dictionary && dictionary[key];
108         if(!message) {
109             message = key;
110         }
111         return message;
112     }
113     
114     
115 };
