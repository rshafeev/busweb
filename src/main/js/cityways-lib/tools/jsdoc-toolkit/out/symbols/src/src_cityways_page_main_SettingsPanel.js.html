 10 cityways.page.main.SettingsPanel = cityways.Class({
 11     
 12     /**
 13      * Список возможных типов маршрутов, по которым ведется поиск,String[]
 14      * Возможные значения: ['c_route_bus', 'c_route_metro',...]
 15      */
 16     availableRouteTypes : null,
 17 
 18     /**
 19      * 
 20      * @param {Object}
 21      *            mainPage
 22      */
 23     initialize : function(options) {
 24         // Найдем в dom все типы маршрутов
 25         this.availableRouteTypes = options.availableRouteTypes;
 26     },
 27     
 28     
 29     getEnabledRouteTypes : function(){
 30        cityways.logger.info(this.availableRouteTypes);
 31        // $("mini_table_transp_list").a
 32         
 33     },
 34    
 35     /**
 36      * 
 37  * @param {String} routeType  Тип маршрута
 38  * @param {Bool} val вкл/выкл
 39  * @param {String} title Надпись
 40      */
 41     enableRouteTypeBtn : function(routeType,val,title){
 42         this.getEnabledRouteTypes();
 43         var resourcePath = cityways.options.ServerHost;
 44         var elem = this._getRouteTypeBtnHtmlElement(routeType);
 45         var img  = elem.find("img").get(0);
 46         var imgFileName = resourcePath  + 'media/css/images/route_types/32/' + routeType;
 47         //elem.title = title; 
 48         if(val == true){
 49             img.src = imgFileName + '_selected.png';
 50         }else
 51         {
 52             img.src = imgFileName + '.png';
 53         }
 54        
 55     },
 56     
 57     _getRouteTypeBtnHtmlElement : function(routeType){
 58          cityways.logger.info($("#cways_menu_route_btn_" + routeType));
 59           return $("#cways_menu_route_btn_" + routeType);
 60     },
 61     
 62     isEnabledRouteTypeBtn : function(routeType){
 63         var elem = this._getRouteTypeBtnHtmlElement(routeType);
 64         var resourcePath = cityways.options.ServerHost;
 65         var img  = elem.find("img").get(0);
 66         var imgFileName = resourcePath  + 'media/css/images/route_types/32/' + routeType;
 67         
 68         if (img.src.indexOf(imgFileName + '.png') != -1) {
 69                 return false;
 70             } else {
 71             return true;
 72         }  
 73     }
 74     
 75     
 76 });
