 10  var cityways =  {
 11     lang : {},
 12     loader : {},
 13     page : {},
 14     type : {},
 15     widget : {},
 16     template : {},
 17     model : {},
 18     helper: {},
 19     map : {},
 20     logger : {},
 21     
 22     /**
 23     * Версия библиотеки
 24     * @type {String}
 25     */
 26     VERSION_NUMBER : "Release 2.13 dev",
 27 
 28      /**
 29      * Function: inherit
 30      * 
 31      * Parameters: C - {Object} the class that inherits P - {Object} the
 32      * superclass to inherit from
 33      * 
 34      * In addition to the mandatory C and P parameters, an arbitrary number of
 35      * objects can be passed, which will extend C.
 36      */
 37      inherit : function(C, P) {
 38         
 39         var F = function() {
 40         };
 41         F.prototype = P.prototype;
 42         C.prototype = new F;
 43         var i, l, o;
 44         for (i = 2, l = arguments.length; i < l; i++) {
 45             o = arguments[i];
 46             if (typeof o === "function") {
 47                 o = o.prototype;
 48             }
 49             cityways.extend(C.prototype, o);
 50             
 51         }
 52         
 53     },
 54 
 55     /**
 56      * [extend description]
 57      * @param  {[type]} destination [description]
 58      * @param  {[type]} source      [description]
 59      * @return {[type]}             [description]
 60      */
 61      extend : function(destination, source) {
 62         destination = destination || {};
 63         if (source) {
 64             for (var property in source) {
 65                 var value = source[property];
 66                 if (value !== undefined) {
 67                     destination[property] = value;
 68                 }
 69             }
 70             
 71             
 72             /**
 73              * IE doesn't include the toString property when iterating over an
 74              * object's properties with the for(property in object) syntax.
 75              * Explicitly check if the source has its own toString property.
 76              */
 77 
 78             /*
 79              * FF/Windows < 2.0.0.13 reports "Illegal operation on WrappedNative
 80              * prototype object" when calling hawOwnProperty if the source
 81              * object is an instance of window.Event.
 82              */
 83 
 84              var sourceIsEvt = typeof window.Event == "function"
 85              && source instanceof window.Event;
 86 
 87              if (!sourceIsEvt && source.hasOwnProperty
 88                 && source.hasOwnProperty("toString")) {
 89                 destination.toString = source.toString;
 90         }
 91     }
 92     return destination;
 93 }
