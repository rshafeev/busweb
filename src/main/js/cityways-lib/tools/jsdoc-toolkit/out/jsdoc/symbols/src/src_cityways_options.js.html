  9 cityways.options = {
 10     
 11     ServerHost : "http://ways.in.ua",
 12     
 13     Theme : "default",
 14     
 15     /**
 16      * Property: ImgPath {String} Set this to the path where control images are
 17      * stored, a path given here must end with a slash. If set to '' (which is
 18      * the default) OpenLayers will use its script location + "img/".
 19      * 
 20      * You will need to set this property when you have a singlefile build of
 21      * 
 22      * Layers that either is not named "OpenLayers.js" or if you move the file
 23      * in a way such that the image directory cannot be derived from the script
 24      * location.
 25      * 
 26      * If your custom OpenLayers build is named "my-custom-ol.js" and the images
 27      * of OpenLayers are in a folder "/resources/external/images/ol" a correct
 28      * way of including OpenLayers in your HTML would be:
 29      * 
 30      * (code) <script src="/path/to/my-custom-ol.js" type="text/javascript"></script>
 31      * <script type="text/javascript"> // tell OpenLayers where the control
 32      * images are // remember the trailing slash OpenLayers.ImgPath =
 33      * "/resources/external/images/ol/"; </script> (end code)
 34      * 
 35      * Please remember that when your OpenLayers script is not named
 36      * "OpenLayers.js" you will have to make sure that the default theme is
 37      * loaded into the page by including an appropriate <link>-tag, e.g.:
 38      * 
 39      * (code) <link rel="stylesheet" href="/path/to/default/style.css"
 40      * type="text/css"> (end code)
 41      */  
 42 	ResourceURI : "/",
 43 	/**
 44 	 * Constant: VERSION_NUMBER
 45 	 */
 46 	VERSION_NUMBER : "Release 2.13 dev",
 47 
 48     /**
 49 	 * Method: _getScriptLocation Return the path to this script. This is also
 50 	 * implemented in OpenLayers.js
 51 	 * 
 52 	 * Returns: {String} Path to this script
 53 	 */
 54 
 55 	_getScriptLocation: function(scriptName) {
 56 	    var regPattern = "(^|(.*?\\/))("+scriptName+"[^\\/]*?\\.js)(\\?|$)";
 57         var r = new RegExp(regPattern),
 58             s = document.getElementsByTagName('script'),
 59             src, m, l = "";
 60         for(var i=0, len=s.length; i<len; i++) {
 61             src = s[i].getAttribute('src');
 62             if(src) {
 63                 m = src.match(r);
 64                 if(m) {
 65                     l = m[1];
 66                     break;
 67                 }
 68             }
 69         }
 70         return l;
 71    },
 72    
 73    getResourcePath : function(){
 74    	 return cityways.options.ResourceURI +"themes/" + cityways.options.Theme + "/";
 75    },
 76    
 77    getBrowserName : function(browser){
 78        var name = null;
 79        $.each(browser, function(i, val) {
 80            if(i != "version")
 81             {
 82                 name = i;
 83                 return false;
 84             }
 85         });
 86        return name;
 87    },
 88    
 89    getCurrentBrowserName : function(){
 90        return cityways.options.getBrowserName($.browser);
 91    }
 92 
 93 };
