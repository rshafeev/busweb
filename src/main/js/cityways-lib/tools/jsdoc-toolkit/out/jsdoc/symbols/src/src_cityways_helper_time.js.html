 12 /**
 13  * @namespace cityways.helper.time
 14  * @description [description]
 15  */
 16 cityways.helper.time = {
 17 
 18 /**
 19  * @secs int[0,3600*24]
 20  * @return
 21  */
 22 secsToLocaleString : function(secs) {
 23 
 24 	var h = parseInt(secs / 3600);
 25 	var m = parseInt((secs - h * 3600) / 60);
 26 	if (h==0)
 27 	{
 28 		return cityways.helper.time.minsOfHourToLocaleString(m);
 29 	}
 30 	return cityways.helper.time.hoursToLocaleString(h) +" " +cityways.helper.time.minsOfHourToLocaleString(m);
 31 	
 32 },
 33 
 34 /**
 35  * @mins int [0,60]
 36  */ minsOfHourToLocaleString  : function(mins) {
 37 
 38 	// минут
 39 	if (mins >= 10 && mins <= 20) {
 40 		return mins + " " + cityways.language.translate("time.minutes10");
 41 	}
 42 
 43 	var ostatok = mins % 10;
 44 	// минута
 45 	if (ostatok <= 1) {
 46 		return mins + " " + cityways.language.translate("time.minute");
 47 	}
 48 	// минуты
 49 	if (ostatok > 1 && ostatok < 5) {
 50 		return mins + " " + cityways.language.translate("time.minutes");
 51 	}
 52 	// минут
 53 	return mins + " " + cityways.language.translate("time.minutes_ru");
 54 
 55 },
 56 
 57 /**
 58  * @mins int [0,24]
 59  */ hoursToLocaleString  : function(hours) {
 60 
 61 	//часов
 62 	if (hours >= 5 && hours <= 20) {
 63 		return hours + " " + cityways.language.translate("time.hours");
 64 	}
 65 	var ostatok = hours % 10;
 66 	//час
 67 	if (ostatok == 1) {
 68 		return hours + " " + cityways.language.translate("time.hour");
 69 	}
 70 
 71 	//часа
 72 	if (ostatok >= 2 && ostatok < 5) {
 73 		return hours + " " + cityways.language.translate("time.hours_ru");
 74 	}
 75 	
 76 	return hours + " " + cityways.language.translate("time.hours");
 77 
 78 	}
 79 
 80 };
