 12 /**
 13  * @namespace cityways.event
 14  */
 15  cityways.event = {
 16 
 17 	/**
 18 	 * [listeners description]
 19 	 * @type {Dictionary<Object,Dictionary<String, Array<cityways.map.EventListener > > >}
 20 	 * @private
 21 	 * @example
 22 	 * var listeners = {
 23 	 * 			"obj1" : {
 24 	 * 				"click" : [
 25 	 * 					{cityways.map.EventListener} object,
 26 	 * 				    {cityways.map.EventListener} object
 27 	 * 				    ],
 28 	 *
 29 	 * 				"drag_end" : [
 30 	 * 					{cityways.map.EventListener} object
 31 	 * 				 	]
 32 	 * 			},
 33 	 * 			"obj2" : {
 34 	 * 				"move" : [
 35 	 * 					{cityways.map.EventListener} object,
 36 	 * 				    {cityways.map.EventListener} object,
 37 	 * 					{cityways.map.EventListener} object
 38 	 * 					]
 39 	 *     		}
 40 	 * };
 41 	 */
 42 	 listeners : {},
 43 
 44 	 /**
 45 	  * [addListener description]
 46 	  * @param {Object}   object   				[description]
 47 	  * @param {String}   evt      				[description]
 48 	  * @param {Function} callback 				[description]
 49 	  * @return {cityways.map.EventListener}    [description]
 50 	  */
 51 	 addListener : function(object, evt, callback) {
 52 	 	var listeners = cityways.event.listeners;
 53 	 	if(object[evt]){
 54 	 		var listener = new cityways.EventListener(object, evt, callback);
 55 	 		if(cityways.event.listeners[object] == undefined){
 56 	 			cityways.event.listeners[object] = {};
 57 	 			cityways.event.listeners[object][evt] = [];
 58 	 		}
 59 	 		else 
 60 	 		if(cityways.event.listeners[object][evt] == undefined){
 61 	 			cityways.event.listeners[object][evt] = [];
 62 	 		}
 63 
 64 	 		cityways.event.listeners[object][evt].push(listener);
 65 	 		return listener;
 66 	 	}
 67 	 	return null;
 68 
 69 	 },
 70 
 71 	 /**
 72 	  * [removeListener description]
 73 	  * @param  {cityways.map.EventListener} listener [description]
 74 	  */
 75 	 removeListener : function(listener) {
 76 	 	var listeners = cityways.event.listeners;
 77 	 	if ( listeners[listener.object] && listeners[listener.object][listener.evt]) {
 78 	 		var objListeners = listeners[listener.object][listener.evt];
 79 	 		for(var i=0; i < objListeners.length; i++){
 80 	 			if(objListeners[i] == listener){
 81 	 				objListeners.remove(i);
 82 	 				return;
 83 	 			}
 84 	 		}
 85 	 	}
 86 	 },
 87 
 88 	 /**
 89 	  * [triggerEvent description]
 90 	  * @param  {Object} object [description]
 91 	  * @param  {[type]} evt    [description]
 92 	  * @param  {[type]} args   [description]
 93 	  * @return {[type]}        [description]
 94 	  */
 95 	 triggerEvent : function(object, evt, args) {
 96 	 	var listeners = cityways.event.listeners;
 97 	 	if(listeners[object] && listeners[object][evt]){
 98 	 		var objListeners = listeners[object][evt];
 99 	 		for(var i = 0; i < objListeners.length; i++){
100 	 			objListeners[i].callback(args);
101 	 		}
102 	 	}
103 	 } 
104 
105 
106 };
