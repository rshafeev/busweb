  8 cityways.logger.setLevelMode  = function(level, val){
  9  		if(level == "INFO_LEVEL"){
 10  			if(val == true)
 11  				cityways.logger.info  = console.log.bind(console);
 12  			else
 13  				cityways.logger.info = function(){};
 14 		}
 15  		if(level == "DEBUG_LEVEL"){
 16  			if(val == true)
 17  				cityways.logger.debug  = console.info.bind(console);
 18  			else
 19  				cityways.logger.debug = function(){};
 20 		}
 21 		 if(level == "WARN_LEVEL"){
 22  			if(val == true)
 23  				cityways.logger.warn  = console.warn.bind(console);
 24  			else
 25  				cityways.logger.warn = function(){};
 26 		}
 27 		cityways.logger[level] = val;
 28 }; 
 29 
 30 (function() {
 31 	cityways.logger.setLevelMode("INFO_LEVEL",true);
 32 	cityways.logger.setLevelMode("DEBUG_LEVEL",true);
 33 	cityways.logger.setLevelMode("WARN_LEVEL",true);
 34 })();
