  6 /**
  7  * Статический класс, хранит общие статические свойства
  8  */
  9 cityways.template.html = {
 10 
 11 	_templates : {},
 12 
 13 	/**
 14 	 * Возвращает html шаблон со словаря _templates. Если такой не сущействует,
 15 	 * пытается загрузить его с сервера Функция ассинхронная,
 16 	 * 
 17 	 * @param {Object}
 18 	 *            templateName
 19 	 * @param {Object}
 20 	 *            callback
 21 	 */
 22 	get : function(templateName, callback) {
 23 		var templates = cityways.template.HtmlTemplates._templates;
 24 		if (templates[templateName] === undefined) {
 25 			var loader = new cityways.loader.TemplatesLoader();
 26 			var fileName = cityways.template.HtmlTemplates
 27 					._getFileByTemplateName(templateName);
 28 			loader.loadHtmlTemplates(fileName, function(__templates) {
 29                         cityways.Util.extend(templates,__templates);
 30                         callback(templates[templateName]);
 31             		});
 32 		} else {
 33 			callback(templates[templateName]);
 34 		}
 35 
 36 	},
 37 
 38 	_getFileByTemplateName : function(templateName) {
 39 		// templateName = template-<file_name>-<name>
 40 		return "main";
 41 	}
 42 
 43 };
