 18 /**
 19  * [MainPage description]
 20  * @class cityways.page.MainPage
 21  */
 22 cityways.page.MainPage = cityways.Class({
 23 
 24 	/**
 25 	 * Видимость правой панели
 26 	 */
 27 	/**
 28 	 * Видимость правой панели.
 29 	 * @private
 30 	 * @fieldOf cityways.page.MainPage.prototype
 31 	 * @default false
 32 	 * @typedef {bool}
 33 	 * @type {bool}
 34 	 */
 35 	rightPanelVisible : null,
 36 	
 37 	/**
 38 	 * cityways.Pages.MainPage.RouteTypesPanel Панель "Виды транспорта"
 39 	 */
 40 	settingsPanel : null,
 41 
 42 	widgetEventHandlers : null,
 43 
 44 	currentCity : null,
 45 
 46 	/**
 47 	 * [initialize description]
 48 	 * @constructor
 49 	 * @this cityways.page.MainPage
 50 	 * @class cityways.page.MainPage
 51 	 * @param  {[type]} options [description]
 52 	 * @return {[type]}         [description]
 53 	 */
 54 	initialize : function(options) {
 55 		cityways.log(this);
 56 		cityways.log('MainPage was initialized!!');
 57 		//alert("Hello");
 58 		this.currentCity = options.currentCity;
 59 		this.widgetEventHandlers = new cityways.page.main.WidgetEventHandlers();
 60 		this.settingsPanel = new cityways.page.main.SettingsPanel({
 61 					availableRouteTypes : options.routeTypes
 62 				});
 63 		
 64 		this.visibleRightPanel(false);
 65 		
 66 	},
 67 
 68 	/**
 69 	 * 
 70 	 * Возвращает текущий город.
 71 	 * @methodOf cityways.page.MainPage.prototype
 72 	 * @return {[type]} [description]
 73 	 */
 74 	getCurrentCity : function() {
 75 		return this.currentCity;
 76 	},
 77 
 78 	getWidgetEventHandlers : function() {
 79 		return this.widgetEventHandlers;
 80 	},
 81 
 82 	getSettingsPanel : function() {
 83 		return this.settingsPanel;
 84 	},
 85 
 86 	getPathsOptions : function() {
 87 		var city = this.getCurrentCity();
 88 		var isTransitions = true;
 89 		var alg_type = "c_time";
 90 		var usage_routeTypes = [];
 91 
 92 		if (isTransitions == true) {
 93 			usage_routeTypes.push({
 94 						discount : 1.0,
 95 						route_type_id : "c_route_transition"
 96 					});
 97 		}
 98 
 99 		usage_routeTypes.push({
100 					discount : 1.0,
101 					route_type_id : "c_route_station_input"
102 				});
103 		usage_routeTypes.push({
104 					discount : 1.0,
105 					route_type_id : "c_route_station_output"
106 				});
107 		usage_routeTypes.push({
108 					discount : 0.5,
109 					route_type_id : "c_route_metro"
110 				});
111 		usage_routeTypes.push({
112 					discount : 1.0,
113 					route_type_id : "c_route_bus"
114 				});
115 		usage_routeTypes.push({
116 					discount : 1.0,
117 					route_type_id : "c_route_trolley"
118 				});
119 
120 		var opts = {
121 			cityID : city.id,
122 			p1 : {
123 				lat : 50.0253640226659,
124 				lon : 36.3350757963562
125 			},
126 			p2 : {
127 				lat : 50.0353179327227,
128 				lon : 36.2199825018311
129 			},
130 			outTime : {
131 				dayID : 'c_Sunday',
132 				timeStart : 1000 * (10 * 60 * 60 + 10 * 60)
133 			},
134 			algStrategy : alg_type,
135 			usageRouteTypes : usage_routeTypes
136 		};
137 		return opts;
138 
139 	},
140 
141 	/**
142 	 * 
143 	 * @param {bool}
144 	 *            true: show, false : hide
145 	 */
146 	visibleRightPanel : function(value) {
147 		 var ResourceURI = cityways.Basic.ResourceURI;
148 		if(this.rightPanelVisible == value)
149 			return;
150 		if (value == true) {
151 			$("#map_canvas").width('68%').css({
152 						cursor : "auto",
153 						backgroundColor : "rgb(226, 226, 226)"
154 					});
155 			
156 			document.img.src = ResourceURI + 'themes/default/images/arrow_right.png';
157 					
158 		} else {
159 			$("#map_canvas").width('98.5%').css({
160 						cursor : "auto",
161 						backgroundColor : "rgb(226, 226, 226)"
162 					});
163 					
164 			document.img.src =ResourceURI + 'themes/default/images/arrow_left.png';
165 		}
166 		this.rightPanelVisible = value;
167 		
168 	}
169 });
