  7 /**
  8  * [MainPage description]
  9  * @class cityways.page.MainPage
 10  */
 11 cityways.page.MainPage = cityways.Class({
 12 
 13 	/**
 14 	 * Видимость правой панели
 15 	 */
 16 	/**
 17 	 * Видимость правой панели.
 18 	 * @fieldOf cityways.page.MainPage.prototype
 19 	 * @default false
 20 	 * @typedef {bool}
 21 	 * @type {bool}
 22 	 */
 23 	rightPanelVisible : null,
 24 	
 25 	/**
 26 	 * cityways.Pages.MainPage.RouteTypesPanel Панель "Виды транспорта"
 27 	 */
 28 	settingsPanel : null,
 29 
 30 	widgetEventHandlers : null,
 31 
 32 	currentCity : null,
 33 
 34 	/**
 35 	 * [initialize description]
 36 	 * @constructor
 37 	 * @this cityways.page.MainPage
 38 	 * @class cityways.page.MainPage
 39 	 * @param  {[type]} options [description]
 40 	 * @return {[type]}         [description]
 41 	 */
 42 	initialize : function(options) {
 43 		cityways.log(this);
 44 		cityways.log('MainPage was initialized!!');
 45 		
 46 		this.currentCity = options.currentCity;
 47 		this.widgetEventHandlers = new cityways.page.main.WidgetEventHandlers();
 48 		this.settingsPanel = new cityways.page.main.SettingsPanel({
 49 					availableRouteTypes : options.routeTypes
 50 				});
 51 		
 52 		this.visibleRightPanel(false);
 53 	},
 54 
 55 	/**
 56 	 * 
 57 	 * Возвращает текущий город.
 58 	 * @methodOf cityways.page.MainPage.prototype
 59 	 * @return {[type]} [description]
 60 	 */
 61 	getCurrentCity : function() {
 62 		return this.currentCity;
 63 	},
 64 
 65 	getWidgetEventHandlers : function() {
 66 		return this.widgetEventHandlers;
 67 	},
 68 
 69 	getSettingsPanel : function() {
 70 		return this.settingsPanel;
 71 	},
 72 
 73 	getPathsOptions : function() {
 74 		var city = this.getCurrentCity();
 75 		var isTransitions = true;
 76 		var alg_type = "c_time";
 77 		var usage_routeTypes = [];
 78 
 79 		if (isTransitions == true) {
 80 			usage_routeTypes.push({
 81 						discount : 1.0,
 82 						route_type_id : "c_route_transition"
 83 					});
 84 		}
 85 
 86 		usage_routeTypes.push({
 87 					discount : 1.0,
 88 					route_type_id : "c_route_station_input"
 89 				});
 90 		usage_routeTypes.push({
 91 					discount : 1.0,
 92 					route_type_id : "c_route_station_output"
 93 				});
 94 		usage_routeTypes.push({
 95 					discount : 0.5,
 96 					route_type_id : "c_route_metro"
 97 				});
 98 		usage_routeTypes.push({
 99 					discount : 1.0,
100 					route_type_id : "c_route_bus"
101 				});
102 		usage_routeTypes.push({
103 					discount : 1.0,
104 					route_type_id : "c_route_trolley"
105 				});
106 
107 		var opts = {
108 			cityID : city.id,
109 			p1 : {
110 				lat : 50.0253640226659,
111 				lon : 36.3350757963562
112 			},
113 			p2 : {
114 				lat : 50.0353179327227,
115 				lon : 36.2199825018311
116 			},
117 			outTime : {
118 				dayID : 'c_Sunday',
119 				timeStart : 1000 * (10 * 60 * 60 + 10 * 60)
120 			},
121 			algStrategy : alg_type,
122 			usageRouteTypes : usage_routeTypes
123 		};
124 		return opts;
125 
126 	},
127 
128 	/**
129 	 * 
130 	 * @param {bool}
131 	 *            true: show, false : hide
132 	 */
133 	visibleRightPanel : function(value) {
134 		 var ResourceURI = cityways.Basic.ResourceURI;
135 		if(this.rightPanelVisible == value)
136 			return;
137 		if (value == true) {
138 			$("#map_canvas").width('68%').css({
139 						cursor : "auto",
140 						backgroundColor : "rgb(226, 226, 226)"
141 					});
142 			
143 			document.img.src = ResourceURI + 'themes/default/images/arrow_right.png';
144 					
145 		} else {
146 			$("#map_canvas").width('98.5%').css({
147 						cursor : "auto",
148 						backgroundColor : "rgb(226, 226, 226)"
149 					});
150 					
151 			document.img.src =ResourceURI + 'themes/default/images/arrow_left.png';
152 		}
153 		this.rightPanelVisible = value;
154 		
155 	}
156 });
