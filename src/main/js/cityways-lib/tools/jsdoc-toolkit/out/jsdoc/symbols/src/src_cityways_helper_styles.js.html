 13  /**

 15  * @description Включает в себя вспомогательные функции для работы со стилями.

 17  cityways.helper.styles = {
 18 
 19 	/**
 20  	* Подключает css файл к текущему html документу
 21  	* @param  {String} cssFileName Путь к css файлу
 22  	* @return {void}            
 23  	*/
 24  	include : function(cssFileName) {
 25  		var css = "<link rel=\"stylesheet\" href=\"" + cssFileName + "\" type=\"text/css\">";
 26  		document.write(css);
 27  	},
 28 
 29 	/**
 30 	 * Подключает css файл к текущему html документу в зависимости от  браузера клиента
 31 	 * @param {String} cssFiles Директория, в которой ле к css файлам
 32 	 * @param {Object} cssOptions Хранит связь css файлов и поддерживаемых ими браузеров. 
 33 	 * @return {String} Путь к подключенному css файлу
 34 	 * @example Приведем пример выбора нужного css файла в зависимости от  браузера клиента
 35 	 * var cssFiles = [
 36 	 *         {
 37 	 *             name : "style.css"
 38 	 *         },
 39 	 *         {
 40 	 *             name : "style_ie7.css",
 41      *             browsers : {
 42      *                          "msie" : {min : 7.0}
 43      *                        },
 44 	 *         {
 45 	 *             name : "style_ff.css",
 46 	 *             browsers : {
 47 	 *                           "mozilla" :  {min : 7.0, max : 16.0}
 48 	 *                        }
 49 	 *         }];
 50 	 * var cssFilename = cityways.helper.styles.includeCSSFile("media/css",cssFiles);
 51 	 * // cssFilename = "media/css/style.css", если текущий браузер не ff и ie
 52 	 * // cssFilename = "media/css/style_ie7.css", если текущий браузер ie7+
 53 	 * // cssFilename = "media/css/style_ff.css", если текущий браузер ff версии [7.0, 16.0]
 54 	 */
 55 	 includeCSSFile : function(filePath, cssFiles) {
 56 	 	var browser = {
 57 	 		name : cityways.util.browser(),
 58 	 		version : cityways.util.browserVersion()
 59 	 	};
 60 	 	var fileName =  cityways.helper.Styles._selectCSSFile(cssOptions,browser);
 61 	 	if(fileName == null)
 62 	 		throw new Error("includeCSSFile() was not found css file");
 63 	 	cityways.helper.Styles.include(filePath + fileName);
 64 	 	return filePath + fileName;
 65 	 },
 66 
 67 	/**
 68   	* Выбирает css файл в зафисимости от текущего браузера.
 69   	* @private
 70   	* @param  {Array}  cssFiles  		 Элементы массива хранят в себе названия сss-файлов и браузеры, которые они поддерживают.
 71  	* @param  {Object} browser 			 Информация о браузере. Хранит версию и название текущего браузера.
 72   	* Структура объекта: <br>
 73   	* {<br>
 74   	* 	name    : {String} <br>
 75   	* 	version : {Number} <br>
 76   	* }<br>
 77  	* @param  {String} browser.name      Тип браузера("msie", "mozilla").
 78  	* @param  {Number} browser.version   Версия браузера.
 79  	* @return {String} 					 Название css файла. 
 80  	* @example 
 81  	* // Пример входных параметров:
 82  	* 
 83  	* var browser  = {
 84  	*      name    : "msie",
 85  	*      version : 7.0
 86  	* };
 87  	* 
 88  	* var cssFiles = [
 89 	*         {
 90 	*             name : "style.css"
 91 	*         },
 92 	*         {
 93 	*             name : "style_ie7.css",
 94     *             browsers : {
 95     *                          "msie" : {min : 7.0}
 96     *                        },
 97 	*         {
 98 	*             name : "style_ff.css",
 99 	*             browsers : {
100 	*                           "mozilla" :  {min : 7.0, max : 16.0}
101 	*                        }
102 	*         }];
103 	* // Вызов функции:
104 	* 
105 	* var selectedFile = cityways.styles._selectCSSFile(cssFiles,browser);
106  	*/
107  	_selectCSSFile : function(cssFiles,browser) {
108  		var defaultFile = null;
109  		var selectedFile = null;
110  		
111  		for(var i=0; i < cssFiles.length; i++ ){
112 
113  			if(cssFiles[i].browsers == undefined &&   cssFiles[i].name != undefined){
114  				defaultFile = cssFiles[i].name;
115  			}
116  			else if (cssFiles[i].browsers != undefined){
117 
118  				var cssBrowser = cssFiles[i].browsers[browser.name];
119  				if(cssBrowser != undefined){
120  					var min_v = cssBrowser.min;
121  					var max_v = cssBrowser.max;
122  					cityways.logger.debug("find",cssFiles[i]);
123  		 				if( (min_v == undefined && max_v == undefined) ||
124  							(min_v == undefined && max_v >= browser.version) ||
125  							(max_v == undefined && min_v <= browser.version) ||
126  							(min_v <= browser.version && max_v >= browser.version)){
127  								selectedFile = cssFiles[i].name;
128  						}
129  				}
130  			}
131  		}
132  		cityways.logger.debug(defaultFile);
133  		if(selectedFile != undefined)
134  			return selectedFile;
135  		return defaultFile;
136  	}
137  };
