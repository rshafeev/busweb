 10 
 11 			/**
 12 			 * 
 13 			 * @param {Object}

 15 			 */
 16 			initialize : function() {
 17 
 18 			},
 19 
 20 			onRouteTypeClick : function(self, routeType, includeTitle,
 21 					excludeTitle) {
 22 				var page = cityways.Page.Current;
 23 				if (page.getSettingsPanel().isEnabledRouteTypeBtn(routeType) == true) {
 24 					page.getSettingsPanel().enableRouteTypeBtn(routeType,
 25 							false, includeTitle);
 26 				} else {
 27 					console.log("false");
 28 					page.getSettingsPanel().enableRouteTypeBtn(routeType, true,
 29 							excludeTitle);
 30 				}
 31 
 32 			},
 33 			_onMakeRightPanelContent : function(self, data, templates) {
 34 				var headersTemplate = templates["template-main-waysPanelHeader"];
 35 				var infoTemplate = templates["template-main-waysPanelInfo"];
 36 				var paths = data.paths;
 37 				var htmlBody = "";
 38 				for (var i = 0; i < paths.length; i++) {
 39 					var path = new cityways.model.Path(paths[i]);
 40 					cityways.Console.log(path.getFullCost());
 41 					var params = {
 42 						index : i + 1,
 43 						locale : cityways.Language.translate,
 44 						cost : path.getFullCost(),
 45 						time: path.getWalkingTime()
 46 					};
 47 					htmlBody = htmlBody + headersTemplate(params);
 48 				}
 49 				$('#panel_data').html(htmlBody);
 50 			},
 51 
 52 			_onLoadedPaths : function(self, e) {
 53 				var data = e.data;
 54 				var templates = {};
 55 				var loadedTemplates = 0;
 56 				var templatesCount = 2;
 57 				var _this = this;
 58 
 59 				cityways.template.HtmlTemplates.get(
 60 						"template-main-waysPanelHeader", function(template) {
 61 							loadedTemplates++;
 62 							templates["template-main-waysPanelHeader"] = template;
 63 							if (loadedTemplates == templatesCount) {
 64 								_this._onMakeRightPanelContent(_this, data,
 65 										templates);
 66 							}
 67 						});
 68 
 69 				cityways.template.HtmlTemplates.get(
 70 						"template-main-waysPanelInfo", function(template) {
 71 							loadedTemplates++;
 72 							templates["template-main-waysPanelInfo"] = template;
 73 							if (loadedTemplates == templatesCount) {
 74 								_this._onMakeRightPanelContent(_this, data,
 75 										templates);
 76 							}
 77 						});
 78 
 79 			},
 80 
 81 			onFindBtnClick : function() {
 82 				var page = cityways.Page.Current;
 83 				var options = page.getPathsOptions();
 84 
 85 				var el = document.getElementById('ways_panel');
 86 				if (options == null)
 87 					return;
 88 				el.style.display = 'block';
 89 				var loadGif = cityways.Basic.getResourcePath()
 90 						+ "images/load.gif";
 91 				$('#panel_data').html("<div class='loader_text'><img src='"
 92 						+ loadGif + "'/></div>");
 93 				$('#panel_scrollbar').tinyscrollbar_update();
 94 				page.visibleRightPanel(true);
 95 				var _this = this;
 96 				cityways.Console.log("onFindBtnClick");
 97 				var loader = new cityways.loader.PathsLoader();
 98 				loader.findShortestPaths(options, function(e) {
 99 							cityways.Console.log("findShortestPaths");
100 							_this._onLoadedPaths(_this, e);
101 						});
102 
103 			},
