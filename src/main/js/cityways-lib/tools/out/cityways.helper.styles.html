<private, static> _selectCSSFile(cssFiles, browser) → {String}

Выбирает css файл в зафисимости от текущего браузера.
Parameters:
Name Type Description
cssFiles Array Элементы массива хранят в себе названия сss-файлов и браузеры, которые они поддерживают.
browser Object Информация о браузере. Хранит версию и название текущего браузера. Структура объекта:
{
name : {String}
version : {Number}
}

Name Type Description

Example
 
// Пример входных параметров:

var browser  = {
     name    : "msie",
     version : 7.0
};

var cssFiles = [
        {
            name : "style.css"
        },
        {
            name : "style_ie7.css",
            browsers : {
                         "msie" : {min : 7.0}
                       },
        {
            name : "style_ff.css",
            browsers : {
                          "mozilla" :  {min : 7.0, max : 16.0}
                       }
        }];
// Вызов функции:

var selectedFile = cityways.styles._selectCSSFile(cssFiles,browser);

Type

Example
 Приведем пример выбора нужного css файла в зависимости от  браузера клиента
var cssFiles = [
        {
            name : "style.css"
        },
        {
            name : "style_ie7.css",
            browsers : {
                         "msie" : {min : 7.0}
                       },
        {
            name : "style_ff.css",
            browsers : {
                          "mozilla" :  {min : 7.0, max : 16.0}
                       }
        }];
var cssFilename = cityways.helper.styles.includeCSSFile("media/css",cssFiles);
// cssFilename = "media/css/style.css", если текущий браузер не ff и ie
// cssFilename = "media/css/style_ie7.css", если текущий браузер ie7+
// cssFilename = "media/css/style_ff.css", если текущий браузер ff версии [7.0, 16.0]
